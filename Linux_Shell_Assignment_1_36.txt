#!/bin/bash

7. Accept Username
#!/bin/bash

8. Display Username and Password
#!/bin/bash
read -p "Enter username: " username
read -s -p "Enter password: " password
echo
echo "Username: $username"
echo "Password: $password"

9. Add Two Numbers Using Command-Line Arguments
#!/bin/bash

#!/bin/bash

11. Display All Command-Line Arguments
#!/bin/bash

12. Count Total Arguments Passed
#!/bin/bash

13. Concatenate Two Strings
#!/bin/bash

14. Swap Two Numbers
#!/bin/bash
a=10
b=20
temp=$a
a=$b
b=$temp
echo "a=$a b=$b"

15. Square and Cube of Number
#!/bin/bash

16. Display All .txt Files

17. Display Hidden Files
#!/bin/bash

18. Count Number of Files in Current Directory
#!/bin/bash
find . -maxdepth 1 -type f | wc -l

19. Check Whether File Exists
#!/bin/bash
read -p "Enter file name: " file
if [ -f "$file" ]
then
echo "File exists"
else
echo "File does not exist"
fi

20. Check Whether Directory Exists
#!/bin/bash
read -p "Enter directory name: " dir
if [ -d "$dir" ]
then
echo "Directory exists"
else
echo "Directory does not exist"
fi

21. Backup Files Into Another Directory
#!/bin/bash
read -p "Enter source directory: " src
read -p "Enter destination directory: " dest
cp -r "$src"/* "$dest"

22. Count Lines in File
#!/bin/bash

23. Count Words in File
#!/bin/bash
read -p "Enter file name: " file

24. Count Characters in File
#!/bin/bash

25. Search Word in File Using grep
#!/bin/bash
read -p "Enter file name: " file

grep "$word" "$file"

26. Display First and Last Line of File
#!/bin/bash
read -p "Enter file name: " file

27. Append Text Into File
#!/bin/bash
read -p "Enter file name: " file
read -p "Enter text: " text
echo "$text" >> "$file"

28. Create System Report File
#!/bin/bash
echo "Username: $(whoami)" > report.txt
echo "Hostname: $(hostname)" >> report.txt
echo "Date: $(date)" >> report.txt
echo "Current Directory: $(pwd)" >> report.txt

31. Display Memory Usage

32. Display Disk Usage

33. Monitor Running Processes
#!/bin/bash

34. Arithmetic Operations Using expr
#!/bin/bash
read -p "Enter first number: " a
read -p "Enter second number: " b
echo "Addition = $(expr $a + $b)"
echo "Subtraction = $(expr $a - $b)"
echo "Multiplication = $(expr $a \* $b)"
echo "Division = $(expr $a / $b)"

35. Display Process ID of Current Shell
#!/bin/bash

36. Print Multiplication Table
#!/bin/bash
read -p "Enter number: " num
for ((i=1;i<=10;i++))
do
echo "$num x $i = $((num*i))"
done
